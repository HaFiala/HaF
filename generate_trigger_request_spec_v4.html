«Function»Generate trigger request

Generate trigger request — Spec v4

Description: The function creates physical trigger file in proper folder and moves request from candidate to triggered

Technically:

1)

<Trigger file name determination>

<if {candidate.requested_by} starts with string 'LANDING' >

parse {landKey} from {candidate.requested_by} like a string between curle brackets "{}"

{triggerFileName} :=

select

case when lnd.sandbox_code ='' then 'SBXDEFAULT' else lnd.sandbox_code end || '_' || lnd.full_file_name || '_' || @trfId || '_' || to_char(@triggerCandid.business_date, 'YYYYMMDD') || '_' || coalesce(@occ2trigger,1) || '.trigger' as triggerFileName

from [table] rt_landing_full

where rt_landing_key = { landKey }

</if {candidate.requested_by} starts with string 'LANDING' >

<if {candidate.requested_by} doesn't start with string 'LANDING' >

{triggerFileName} = {candidate.uacWorkflowName} || '_' || {candidate.transformation_id} || '_' || FORMAT({candidate.businessDate}, 'YYYYMMDD') || '_' || {occ2Trigger} || '.trigger'

</if {candidate.requested_by} doesn't start with string 'LANDING' >

</Trigger file name determination>

2)

<Agent determination>

NOTE: folder determination will be replaced by agent determination {agentURL}

</Agent determination>

<Trigger folder determination>

{specific_trigger_folder} = empty

write into [register] triggerFolderExceptions result from

select

from [table] ot_dynamic_property

SELECT TRIM(elem) AS pattern

FROM ot_dynamic_property,

unnest(string_to_array(property_value, ',')) AS elem

WHERE property_name = 'batch.server.exception'

AND TRIM(elem) <> '';

read from [register] triggerFolderExceptions where {triggerFolderException.pattern} = {candidate.uacWorkflowName}

if record is found then {specific_trigger_folder} :=

select

specific_trigger_folder

from [table] dt_trigger_folder_def

where UPPER(target_layer) = {candidate.targetLayer}

if {specific_trigger_folder} is empty then {triggerFileFolder} = {candidate.defaultTriggerFolder}

if {specific_trigger_folder} is not empty then {triggerFileFolder} = {specific_trigger_folder}

</Trigger folder determination>

3)

triggers «Interface»[API] receive-trigger

<if response is OK (200)>

update [table] rt_event_request_full

set deleted_flag = 0, request_status = 'REQT', updated_dt = current_timestamp, trigger_file_name = {triggerFileName}, occurrence = {occ2trigger}

where rt_event_request_key = {candidate.eventRequestKey}

</if response is OK (200)>

<if response is NOK>

update [table] rt_event_request_queue

set updated_datetime = current_timestamp, trigger_file_name = 'BAD_AGENT_RESPONSE ' & {agentURL}

where rt_event_request_key = {candidate.eventRequestKey}

</if response is NOK>
